63 extern inline void serial_init (void)
64 {
65  ((unsigned volatile short*) &_io_ports[M6812_SCBD])[0] = M6812_DEF_BAUD;
66 
67  /* Setup character format 1 start, 8-bits, 1 stop. */

113 extern inline void serial_send (char c)
114 {
115  serial_flush ();
116  _io_ports[M6812_SCDR] = c;

130 extern inline uint8_t serial_recv (void)
131 {
